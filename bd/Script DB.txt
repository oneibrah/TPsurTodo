CREATE TABLE Utilisateur(idUser int PRIMARY KEY NOT NULL AUTO_INCREMENT, nomUser varchar(45) NOT NULL, prenomUser varchar(45) NOT NULL,typeUser varchar(20) NOT NULL, pseudo varchar(45) UNIQUE, login varchar(45) UNIQUE NOT NULL, PASSWORD varchar(45) NOT NULL);
CREATE TABLE Projet(idProjet int PRIMARY KEY NOT NULL AUTO_INCREMENT, nomProjet varchar(45) NOT NULL, date_debut DATE NOT NULL, date_fin DATE NOT NULL, etat varchar(20) NOT NULL, idUser int, FOREIGN KEY(idUser) REFERENCES utilisateur(idUser));
CREATE TABLE Etape(idPhase int PRIMARY KEY NOT NULL AUTO_INCREMENT, nomPhase varchar(45) NOT NULL, date_debut DATE NOT NULL, date_fin DATE NOT NULL, etat varchar(20) NOT NULL, idProjet int, FOREIGN KEY(idProjet) REFERENCES projet(idProjet));
CREATE TABLE Tache(idTache int PRIMARY KEY NOT NULL AUTO_INCREMENT, nomTache varchar(45) NOT NULL, descriptionTache varchar(255), date_debut DATE NOT NULL, date_fin DATE NOT NULL, etat varchar(20) NOT NULL, idProjet int, idPhase int, FOREIGN KEY(idProjet) REFERENCES projet(idProjet), FOREIGN KEY(idPhase) REFERENCES etape(idPhase));
CREATE TABLE Tache_Utilisateur(idTacheUser int PRIMARY KEY NOT NULL AUTO_INCREMENT, idUser int, idTache int, FOREIGN KEY(idUser) REFERENCES utilisateur(idUser), FOREIGN KEY(idTache) REFERENCES tache(idTache));
